```bash
# 检查所有 subprocess.run 是否有 encoding="utf-8"

# 检查所有 open() 是否有 encoding
grep -rn 'open(' src/ahadiff/ --include='*.py' | grep -v __pycache__ | grep -v 'encoding' | grep -v '#' | grep -v "'rb'" | grep -v "'wb'"

# 检查 Windows reparse point 重复代码
grep -rn '_has_windows_reparse_point\|st_file_attributes.*0x400' src/ahadiff/ --include='*.py'

# 检查 WSL2 检测
grep -rn 'is_wsl2_mnt\|WSL_DISTRO\|WSL_INTEROP' src/ahadiff/ --include='*.py'

# 检查 symlink 防御
grep -rn 'O_NOFOLLOW\|lstat\|S_ISLNK\|is_symlink' src/ahadiff/ --include='*.py' | wc -l

# 检查 pathlib vs os.path
grep -rn 'os\.path\.' src/ahadiff/ --include='*.py' | grep -v __pycache__
```

额外检查：
- Windows 上 `portalocker` 是否可用？
